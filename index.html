export default function ProductKPIDashboard() {

  // Load saved data from localStorage or use defaults

  const [kpiData, setKpiData] = useState(() => {

    try {

      const saved = localStorage.getItem('productKPIData');

      if (saved) {

        const parsedData = JSON.parse(saved);

        // Ensure initiatives exist, add default if missing

        if (!parsedData.initiatives) {

          parsedData.initiatives = [

            { id: 1, name: 'Health Cloud transformation (unify all products)', status: 'green', commentary: '' },

        return parsedData;

    } catch (error) {

      console.log('Could not load saved data');

    }

    return {

      month: 'December',

      year: '2025',

      roadmapCurrent: 85,

      roadmapTarget: 80,

      roadmapChange: 16,

      roadmapColor: 'yellow',

      changeRequestsCurrent: 10,

      changeRequestsTarget: 4,

      changeRequestsChange: 0,

      changeRequestsColor: 'green',

      backlogCurrent: 197,

      backlogTarget: 125,

      backlogChange: 9,

      metric4Current: 0,

      metric4Target: 0,

    try {

      const saved = localStorage.getItem('productKPIPreviousData');

      if (saved) {

        return JSON.parse(saved);

      }

      console.log('No previous data found');

  const [showConfirmModal, setShowConfirmModal] = useState(false);

  const [skipChangeCalc, setSkipChangeCalc] = useState(false);

  const [expandedInitiative, setExpandedInitiative] = useState(null);

  const [editingInitiative, setEditingInitiative] = useState(null);


  // Save current data whenever it changes

  useEffect(() => {

    try {

      localStorage.setItem('productKPIData', JSON.stringify(kpiData));

    } catch (error) {

    }

  }, [kpiData]);


  const getStatusColor = (current, target, lowerIsBetter = false) => {

    if (lowerIsBetter) {

      if (current <= target) return 'green';

      if (current <= target * 1.5) return 'yellow';

      return 'red';

    } else {

      if (current >= target) return 'green';

      if (current >= target * 0.8) return 'yellow';

      return 'red';

    }

  };


  const getProgressPercent = (current, target) => {

    if (current === 'N/A' || target === 'N/A') return 0;

    return Math.min((current / target) * 100, 100);

  };


  const roadmapStatus = kpiData.roadmapCurrent === 'N/A' ? 'gray' : (kpiData.roadmapColor || 'yellow');

  const changeRequestsStatus = kpiData.changeRequestsCurrent === 'N/A' ? 'gray' : (kpiData.changeRequestsColor || getStatusColor(kpiData.changeRequestsCurrent, kpiData.changeRequestsTarget, true));

  const backlogStatus = kpiData.backlogCurrent === 'N/A' ? 'gray' : (kpiData.backlogColor || getStatusColor(kpiData.backlogCurrent, kpiData.backlogTarget));

  const metric4Status = kpiData.metric4Name === 'TBD' || kpiData.metric4Current === 'N/A' ? 'gray' : getStatusColor(kpiData.metric4Current, kpiData.metric4Target);


  const handleInputChange = (field, value) => {

    setKpiData(prev => ({

      ...prev,

      [field]: value

    }));

  };


  const startNewMonth = () => {

    setShowConfirmModal(true);

  };


  const confirmStartNewMonth = () => {

    try {

      if (!skipChangeCalc) {

        // Save current data as previous only if we want to track changes

        localStorage.setItem('productKPIPreviousData', JSON.stringify(kpiData));

        setPreviousData(kpiData);

      } else {

        // Clear previous data so no changes are calculated

        localStorage.removeItem('productKPIPreviousData');

        setPreviousData(null);

        // Reset all change values to 0

        setKpiData(prev => ({

          ...prev,

          roadmapChange: 0,

          changeRequestsChange: 0,

          backlogChange: 0,

          metric4Change: 0

        }));

      }

      

      setShowConfirmModal(false);

      setShowInputs(true);

      setSkipChangeCalc(false); // Reset for next time

    } catch (error) {

      console.error('Save error:', error);

    }

  };


  const cancelStartNewMonth = () => {

    setShowConfirmModal(false);

  };


  const updateInitiativeStatus = (id, newStatus) => {

    setKpiData(prev => ({

      ...prev,

      initiatives: prev.initiatives.map(init => 

        init.id === id ? { ...init, status: newStatus } : init

      )

    }));

  };


  const updateInitiativeCommentary = (id, commentary) => {

    setKpiData(prev => ({

      ...prev,

      initiatives: prev.initiatives.map(init => 

        init.id === id ? { ...init, commentary } : init

      )

    }));

  };


  const updateInitiativeName = (id, name) => {

    setKpiData(prev => ({

      ...prev,

      initiatives: prev.initiatives.map(init => 

        init.id === id ? { ...init, name } : init

      )

    }));

  };


  const toggleExpand = (id) => {

    setExpandedInitiative(expandedInitiative === id ? null : id);

    setEditingInitiative(null);

  };


  // Auto-calculate changes based on previous month

  useEffect(() => {

    if (previousData) {

      setKpiData(prev => ({

        ...prev,

        roadmapChange: Math.round(prev.roadmapCurrent - previousData.roadmapCurrent),

        changeRequestsChange: Math.round(prev.changeRequestsCurrent - previousData.changeRequestsCurrent),

        backlogChange: Math.round(prev.backlogCurrent - previousData.backlogCurrent),

        metric4Change: previousData.metric4Current ? Math.round(prev.metric4Current - previousData.metric4Current) : 0

      }));

    }

  }, [

    kpiData.roadmapCurrent, 

    kpiData.changeRequestsCurrent, 

    kpiData.backlogCurrent,

    kpiData.metric4Current,

    previousData

  ]);


  const exportToPNG = () => {

    // Simple instruction for user

    alert('To save as image:\n1. Take a screenshot of the dashboard\n2. Or use browser print > Save as PDF');

  };

    <div style={{

      fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", "Helvetica Neue", Arial, sans-serif',

      background: 'linear-gradient(135deg, #f5f7fa 0%, #c3cfe2 100%)',

      minHeight: '100vh',

      padding: '40px 20px'

    }}>

            background: 'white',

            borderRadius: '16px',

            maxWidth: '500px',

            boxShadow: '0 20px 60px rgba(0,0,0,0.3)'

          }}>

            <h3 style={{

              fontSize: '20px',

              fontWeight: '700',

              color: '#1a1a1a',

              marginBottom: '16px'

            }}>

              Save Current Period as Baseline?

            </h3>

            <p style={{

              fontSize: '15px',

              color: '#666',

              lineHeight: '1.6',

              marginBottom: '20px'

            }}>

              This will save current data and calculate changes automatically when you enter new values.

            </p>

            

            <label style={{

              display: 'flex',

              alignItems: 'center',

              gap: '8px',

              marginBottom: '24px',

              cursor: 'pointer',

              fontSize: '14px',

              color: '#4b5563'

            }}>

              <input

                type="checkbox"

                checked={skipChangeCalc}

                onChange={(e) => setSkipChangeCalc(e.target.checked)}

                style={{ cursor: 'pointer' }}

              />

              Skip change calculation (use for first entry or when switching periods)

            <div style={{ display: 'flex', gap: '12px', justifyContent: 'flex-end' }}>

              <button

                  padding: '10px 20px',

                  background: '#f3f4f6',

                  color: '#4b5563',

                  fontSize: '14px',

                  cursor: 'pointer'

                }}

                onClick={confirmStartNewMonth}

                  padding: '10px 20px',

              >

                Confirm

              </button>

            </div>

          </div>

      )}

          <div>

            {previousData && (

                display: 'inline-block',

                padding: '8px 16px',

                color: 'white',

                borderRadius: '8px',

              <div style={{

                padding: '16px',

                background: '#f0fdf4',

                borderRadius: '8px',

                marginBottom: '24px',

                border: '1px solid #bbf7d0'

              }}>

                <div style={{ fontSize: '14px', color: '#166534', fontWeight: '600', marginBottom: '4px' }}>

                  ✓ Previous Month Saved - Changes Auto-Calculated

                </div>

                </label>

                  value={`${kpiData.month} ${kpiData.year}`}

                    fontWeight: '600'

              <input type="hidden" value={kpiData.year} />

                  Roadmap Current (%)

                  />

                  <span style={{ fontSize: '12px', color: '#9ca3af' }}>N/A</span>

                </label>

                <input

                  type="number"

                  value={kpiData.roadmapCurrent === 'N/A' ? '' : kpiData.roadmapCurrent}

                  onChange={(e) => handleInputChange('roadmapCurrent', parseFloat(e.target.value) || 0)}

                  disabled={kpiData.roadmapCurrent === 'N/A'}

                    width: '100%',

                  type="number"

                    width: '100%',

                    padding: '10px 12px',

                    border: '2px solid #e5e7eb',

                    borderRadius: '8px',

                    fontSize: '15px'

                        background: color === 'green' ? '#22c55e' : color === 'yellow' ? '#f59e0b' : '#ef4444',

                  ))}

  const isNA = current === 'N/A' || current === '--';

  

  const statusColors = {

    green: { bg: '#22c55e', shadow: 'rgba(34, 197, 94, 0.2)', progressBg: 'linear-gradient(90deg, #667eea 0%, #764ba2 100%)', changeBg: '#f0fdf4', changeColor: '#22c55e' },

  };


  // Use gray for placeholder or N/A values, otherwise use the status color

    <div style={{

      background: (isPlaceholder || isNA) ? '#f9fafb' : 'white',

      padding: '20px',

      }} />


      {/* Header with title and status */}

      <div style={{

      }}>

            lineHeight: '1.2'

          }}>

          </div>

            <div style={{

              fontSize: '9px',

              color: (isPlaceholder || isNA) ? '#9ca3af' : '#6b7280',

            }}>

        <div style={{

          fontSize: '36px',

          fontWeight: '700',

          lineHeight: '1',

          marginBottom: '12px'

        }}>
